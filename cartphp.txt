    <script>
        document.addEventListener('DOMContentLoaded', function () {
            function updateTotal() {
                let total = 0;
                document.querySelectorAll('.cart-item').forEach(item => {
                    const quantity = item.querySelector('.quantity-input').value;
                    const price = parseFloat(item.querySelector('.item-price').dataset.price);
                    const totalPrice = price * quantity;
                    item.querySelector('.item-price').textContent = `₹ ${totalPrice.toFixed(2)}`;
                    total += totalPrice;
                });
                document.getElementById('cart-total').textContent = `Total: ₹ ${total.toFixed(2)}`;
            }

            document.querySelectorAll('.cart-item').forEach(item => {
                item.querySelector('.minus-btn').addEventListener('click', function () {
                    let quantityInput = item.querySelector('.quantity-input');
                    if (quantityInput.value > 1) {
                        quantityInput.value--;
                        updateTotal();
                    }
                });

                item.querySelector('.add-btn').addEventListener('click', function () {
                    let quantityInput = item.querySelector('.quantity-input');
                    quantityInput.value++;
                    updateTotal();
                });

                item.querySelector('.quantity-input').addEventListener('input', updateTotal);
            });

            updateTotal(); // Initial calculation
        });
    </script>
